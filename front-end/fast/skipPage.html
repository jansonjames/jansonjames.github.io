# 页面跳转

这两个方法，onSkipTap定义在页面中，navigationTo定义在全局通用工具函数模块中，这样不用每个页面都写一次。

主要目的是将页面中的跳转逻辑集中到一个跳转函数onSkipTap中, 个人用着还算舒服，当页面有多个点击跳转的时候，不用每个点击跳转都定义一个函数,方便简洁很多。

// 页面
onSkipTap(type, params) {
    let urlObj = {
        login: () => {
            // 稍微复杂的操作
        },
        editUserInfo: '/pages/user-editInfo/user-editInfo',
        me: '/pages/me/me',
    }
    this.$g_util.navigationTo(urlObj, type)
}

// 全局
navigationTo(urlObj, type) {
    if (typeof urlObj[type] === 'string') {
        uni.navigateTo({
            url: urlObj[type]
        })
    } else if (typeof urlObj[type] === 'function') {
        urlObj[type]()
    } else {
        console.error('错误页面路径！');
    }
}
